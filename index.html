📜 Как подключить Supabase (Облачная база данных)

Шаг 1: Перейдите на сайт supabase.com и создайте бесплатный проект (New Project).
Шаг 2: В левом меню проекта перейдите в раздел SQL Editor → New Query. Скопируйте SQL-код ниже, вставьте в окно и нажмите кнопку Run.
supabase_setup.sql 
-- 1. Таблицы портфолио, галереи, настроек и текстов
CREATE TABLE IF NOT EXISTS portfolio (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    category TEXT NOT NULL DEFAULT 'Дизайн персонажа',
    image_url TEXT NOT NULL,
    title TEXT DEFAULT '',
    sort_order INTEGER DEFAULT 0,
    created_at TIMESTAMPTZ DEFAULT now()
);
CREATE TABLE IF NOT EXISTS shop (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    name TEXT NOT NULL DEFAULT 'Готовый арт',
    price TEXT DEFAULT 'Цифровая лицензия доступна',
    image_url TEXT NOT NULL,
    available BOOLEAN DEFAULT true,
    created_at TIMESTAMPTZ DEFAULT now()
);
CREATE TABLE IF NOT EXISTS site_texts (id TEXT PRIMARY KEY, value TEXT NOT NULL);
CREATE TABLE IF NOT EXISTS site_settings (key TEXT PRIMARY KEY, value TEXT NOT NULL);

ALTER TABLE portfolio ENABLE ROW LEVEL SECURITY;
ALTER TABLE shop ENABLE ROW LEVEL SECURITY;
ALTER TABLE site_texts ENABLE ROW LEVEL SECURITY;
ALTER TABLE site_settings ENABLE ROW LEVEL SECURITY;

CREATE POLICY "Public read portfolio" ON portfolio FOR SELECT USING (true);
CREATE POLICY "Public insert portfolio" ON portfolio FOR INSERT WITH CHECK (true);
CREATE POLICY "Public update portfolio" ON portfolio FOR UPDATE USING (true);
CREATE POLICY "Public delete portfolio" ON portfolio FOR DELETE USING (true);

CREATE POLICY "Public read shop" ON shop FOR SELECT USING (true);
CREATE POLICY "Public insert shop" ON shop FOR INSERT WITH CHECK (true);
CREATE POLICY "Public update shop" ON shop FOR UPDATE USING (true);
CREATE POLICY "Public delete shop" ON shop FOR DELETE USING (true);

CREATE POLICY "Public read site_texts" ON site_texts FOR SELECT USING (true);
CREATE POLICY "Public upsert site_texts" ON site_texts FOR ALL USING (true);

CREATE POLICY "Public read site_settings" ON site_settings FOR SELECT USING (true);
CREATE POLICY "Public upsert site_settings" ON site_settings FOR ALL USING (true);

INSERT INTO storage.buckets (id, name, public) VALUES ('artworks', 'artworks', true) ON CONFLICT (id) DO NOTHING;
CREATE POLICY "Public Access Artworks" ON storage.objects FOR SELECT USING (bucket_id = 'artworks');
CREATE POLICY "Public Upload Artworks" ON storage.objects FOR INSERT WITH CHECK (bucket_id = 'artworks');
CREATE POLICY "Public Update Artworks" ON storage.objects FOR UPDATE USING (bucket_id = 'artworks');
CREATE POLICY "Public Delete Artworks" ON storage.objects FOR DELETE USING (bucket_id = 'artworks');
Шаг 3: Перейдите в меню Project Settings → API. Скопируйте Project URL и anon / public key, вставьте их в поля настройки на сайте и нажмите «Сохранить ключи».
